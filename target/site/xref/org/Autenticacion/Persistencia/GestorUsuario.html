12  /**
13   * Clase GestorUsuario. Utilizada como intermediario entre el usuario de la
14   * aplicación y el agente que se conecta a la base de datos para realizar las
15   * distintas operaciones. Es la encargada de realizar todas las consultas SQL de
16   * lectura, eliminación, modificación y adición.
17   *
18   * @author in4mates
19   * @version 1.0
20   */
21  
22  public class GestorUsuario {
23  
24  	/**
25  	 * Array list que contendrá a los distintos usuarios que se lean de
26  	 * la base de datos.
27  	 **/
28  	private ArrayList<Usuario> listaUsuarios;
29  
30  	/**
31  	 * Constructor por defecto. Realiza la creacion de la lista vacía de
32  	 * usuarios.
33  	 */
34  	public GestorUsuario() {
35  		this.listaUsuarios = new ArrayList<Usuario>();
36  	}
37  
38  	/**
39  	 * Método que obtiene la lista de usuarios.
40  	 * @return Lista de los distintos usuarios que se cargaron en la
41  	 * lista con anterioridad.
42  	 */
43  	public ArrayList<Usuario> getListaUsuarios() {
44  		return this.listaUsuarios;
45  	}

47  	/**
48  	 * Método que envía la consula SQL de eliminación al agente, si este
49  	 * realiza la operación de manera correcta se obtendrá un valor de
50  	 * true como resultado de la ejecución del método.
51  	 * @param usuario (Usuario que se va a eliminar de la base de datos).
52  	 * @return Valor a true o a false en función de si la consulta se ha
53  	 * ejecutado de manera correcta o no.
54  	 */
55  	public boolean delete(Usuario usuario) {
56  		boolean correcto = false;
57  		try {
58  			Agente.getAgente().modificar("DELETE FROM usuarios "
59  					+ "WHERE nombre = '"
60  					+ usuario.getNombre() + "'");
61  			correcto = true;
62  		} catch (SQLException e) {
63  			// TODO Auto-generated catch block
64  			correcto = false;
65  		} catch (Exception e) {
66  			// TODO Auto-generated catch block
67  			correcto = false;
68  		}
69  		return correcto;
70  	}

72  	/**
73  	 * Método que envía la consula SQL de modificación al agente, si este
74  	 * realiza la operación de manera correcta se obtendrá un valor
75  	 * de true como resultado de la ejecución del método.
76  	 * @param usuario (Usuario que se va a modificar en la base de datos).
77  	 * @return Valor a true o a false en función de si la consulta se ha
78  	 * ejecutado de manera correcta o no.
79  	 */
80  	public boolean update(Usuario usuario) {
81  		boolean correcto = false;
82  		try {
83  			String SQL = "UPDATE usuarios SET nombre='" 
84  					+ usuario.getNombre() + "', " 
85  					+ "pass='" + usuario.getPassword() 
86  					+ "',tipo='" + usuario.getTipo() + "'" 
87  					+ " WHERE nombre = '"
88  					+ usuario.getNombre() + "' AND pass='"
89  					+ usuario.getPassword() + "'";
90  			Agente.getAgente().modificar(SQL);
91  			correcto = true;
92  		} catch (SQLException e) {
93  			// TODO Auto-generated catch block
94  			// e.printStackTrace();
95  			correcto = false;
96  		} catch (Exception e) {
97  			// TODO Auto-generated catch block
98  			correcto = false;
99  		}
100 		return correcto;
101 	}

103 	/**
104 	 * Método que envía la consula SQL de creación de usuario al agente,
105 	 * si este realiza la operación de manera correcta se obtendrá un
106 	 * valor de true como resultado de la ejecución del metodo.
107 	 * @param usuario (Usuario que se va a añadir a la base de datos).
108 	 * @return Valor a true o a false en función de si la consulta se ha

110 	 */
111 	public boolean insert(Usuario usuario) {
112 		boolean correcto = false;
113 		try {
114 			Agente.getAgente().modificar("INSERT INTO usuarios VALUES ('"
115 					+ usuario.getNombre() + "'," + "'"

117 					+ "','" + usuario.getTipo() + "')");
118 			correcto = true;
119 		} catch (SQLException e) {
120 			// TODO Auto-generated catch block
121 			correcto = false;
122 		} catch (Exception e) {
123 			// TODO Auto-generated catch block
124 			correcto = false;
125 		}
126 		return correcto;
127 	}

129 	/**
130 	 * Método que envía la consula SQL de lectura al agente para comprobar
131 	 * de la existencia un usuario según los datos introducidos, si este
132 	 * realiza la operación de manera correcta se obtendrá un valor de true
133 	 * como resultado de la ejecución del método.
134 	 * @param usuario (Usuario sobre el cual se quieren comprobar sus datos
135 	 * para realizar el login).
136 	 * @return Valor a true o a false en funcion de si la consulta se ha
137 	 *         ejecutado de manera correcta o no.
138 	 */
139 	public boolean login(Usuario usuario) {
140 		ResultSet resultado;
141 		boolean existe = false;
142 		try {
143 			resultado = Agente.getAgente()
144 					.leer("SELECT * FROM usuarios WHERE nombre='"
145 							+ usuario.getNombre() + "'" + " AND pass='"
146 							+ usuario.getPassword() + "'");
147 
148 			while (resultado.next()) {
149 				existe = true;
150 				usuario.setNombre(resultado.getString(1));
151 				usuario.setPassword(resultado.getString(2));
152 				usuario.setTipo(resultado.getString(3));
153 
154 			}
155 
156 		} catch (SQLException e) {
157 			// TODO Auto-generated catch block
158 			existe = false;
159 		} catch (Exception e) {
160 			// TODO Auto-generated catch block
161 			existe = false;
162 		}
163 		return existe;
164 	}

166 	/**
167 	 * Método que envía la consula SQL de lectura al agente para obtener los
168 	 * datos referentes a todos los usuarios, una vez se tengan los datos se
169 	 * iran creando usuarios y estos se añadiran a la lista.
170 	 * @return Valor a true o a false en funcion de si la consulta se ha
171 	 *         ejecutado de manera correcta o no.
172 	 */
173 	public boolean readAll() {
174 		ResultSet resultado;
175 		boolean correcto = false;
176 		try {
177 			resultado = Agente.getAgente().leer("SELECT * FROM usuarios");
178 			Usuario usuario;
179 
180 			while (resultado.next()) {
181 				correcto = true;
182 				usuario = new Usuario(resultado.getString(1),
183 						resultado.getString(2), resultado.getString(3));
184 				listaUsuarios.add(usuario);
185 			}
186 		} catch (SQLException e) {
187 			// TODO Auto-generated catch block
188 			correcto = false;
189 		} catch (Exception e) {
190 			// TODO Auto-generated catch block
191 			correcto = false;
192 		}
193 		return correcto;
194 	}
