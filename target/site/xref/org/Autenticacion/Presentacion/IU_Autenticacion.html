127 	public boolean aceptar() {
128 		boolean autenticado=false;
129 		try {
130 			if(txtNombre.getText().equals("") && passwordField.getText().equals("")) {
131 				JOptionPane.showMessageDialog(null, "No ha introducido usuario y contraseÃ±a", "AutenticaciÃ³n",JOptionPane.WARNING_MESSAGE);
132 			}
133 			else {
134 				String nombre=txtNombre.getText();
135 				String pass = passwordField.getText();
136 				Usuario usuario = new Usuario();
137 				usuario.setNombre(nombre);
138 				usuario.setPassword(pass);
139 				
140 				if(!usuario.autenticarse()) {
141 					JOptionPane.showMessageDialog(null, "Usuario o contraseÃ±a incorrectos", "AutenticaciÃ³n",JOptionPane.WARNING_MESSAGE);
142 				}
143 				else {
144 					autenticado=true;
145 					
146 					JOptionPane.showMessageDialog(null, "Autenticacion correcta", "AutenticaciÃ³n",JOptionPane.DEFAULT_OPTION);
147 					
148 				}
149 			}
150 		}catch(Exception arg0){
151 			autenticado=false;
152 		}
153 		return autenticado;
154 	}
