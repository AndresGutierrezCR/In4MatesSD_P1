9    * Clase ReservaMesas. Utilizada para representar la reserva de cada una de las
10   * mesas dentro del sistema. Cada reserva queda determinada por un identificador
11   * de reserva, un nombre de cliente (el cual hace la reserva), el numero de
12   * comensales a reservar, el turno de comida o cena, la mesa que se va a
13   * reservar, y el número del turno en el que se reservará.
14   *
15   * @author in4mates
16   * @version 1.0
17   */
18  public class Reserva {

21  	 **/
22  	private int idReserva;
23  	/**
24  	 * Nombre del cliente (no usuario) que figura en la reserva.
25  	 **/
26  	private String nombreCliente;
27  	/**
28  	 * Número de comesales que atenderán a la mesa.
29  	 **/
30  	private int comensales;
31  	/**
32  	 * Turno del servicio (comida o cena).
33  	 **/
34  	private String turnoComCen;
35  	/**
36  	 * Número del turno dentro de los turnos principales (1, 2 o 3).
37  	 **/
38  	private int turno;

48  	/**
49  	 * Constructor con 6 parametros. Crea una reserva de mesa a partir de un
50  	 * identificador, un nombre de cliente, un numero de comensales, un turno de
51  	 * comida o cena, un número de turno, y una mesa.
52  	 *
53  	 * @param idReserva     ID de la mesa.
54  	 * @param nombreCliente nombre del cliente que reserva la mesa.
55  	 * @param comensales    numero de comensales de la mesa.
56  	 * @param turnoComCen   turno de comida o cena de la reserva.
57  	 * @param turno         número del turno de la reserva.
58  	 * @param mesa          mesa en la que se realiza la reserva.
59  	 * @throws Exception si el identificador de la reserva es menor que 1.
60  	 * @throws Exception si el número de comensales supera al máximo permitido por
61  	 *                   la mesa.
62  	 * @throws Exception si el número de comensales es menor o igual a 0.
63  	 * @throws Exception si el número del turno es menor o igual a 0.
64  	 * @throws Exception si la longitud del nombre del cliente es menor a 1.
65  	 * @throws Exception si la longitud de la cadena del turno es menor a 1.
66  	 * @throws Exception si el turno no es comida o cena.
67  	 */
68  	public Reserva(int idReserva, String nombreCliente, int comensales, String turnoComCen, int turno, Mesa mesa)
69  			throws Exception {
70  
71  		if (idReserva < 1) {
72  			throw new Exception("El numero de reserva no puede ser negativo");
73  		}
74  
75  		if (comensales > mesa.getComensales()) {
76  			throw new Exception("Superado numero maximo de comensales");
77  		}
78  		if (comensales <= 0) {
79  			throw new Exception("el numero de comensales debe ser positivo");
80  		}
81  		if (turno <= 0) {
82  			throw new Exception("Turno no valido");
83  		}
84  
85  		if (nombreCliente.length() < 1) {
86  			throw new Exception("Nombre de cliente no puede ser vacio");
87  		}
88  		if (turnoComCen.length() < 1) {
89  			throw new Exception("Tunro Comdida/Cena no puede ser invalido");
90  		}
91  		if (!(turnoComCen.equals("comida") || turnoComCen.equals("cena"))) {
92  			throw new Exception("No es un turno valido");
93  		}
94  		this.idReserva = idReserva;
95  		this.nombreCliente = nombreCliente;
96  		this.comensales = comensales;
97  		this.turnoComCen = turnoComCen;
98  		this.turno = turno;
99  		this.mesa = mesa;
100 		this.gestorReserva = new GestorReserva();
101 	}
