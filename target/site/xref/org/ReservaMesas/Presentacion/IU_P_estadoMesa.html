457 	/**
458 	 * Clase BtnActionListener.
459 	 *
460 	 * @author in4mates
461 	 * @version 1.0
462 	 */
463 	private class BtnActionListener implements ActionListener {
464 		/**
465 		 * método actionPerformed.
466 		 *
467 		 * @param e .
468 		 **/
469 		public void actionPerformed(ActionEvent e) {
470 
471 			String boton = e.getActionCommand().toString();
472 			Estados estado;
473 			estado = Estados.LIBRE;
474 
475 			if (boton.equals("Libre")) {
476 				estado = Estados.LIBRE;
477 			} else if (boton.equals("Reservada")) {
478 				estado = Estados.RESERVADA;
479 			} else if (boton.equals("Ocupada")) {
480 				estado = Estados.OCUPADA;
481 			} else if (boton.equals("Pidiendo")) {
482 				estado = Estados.PIDIENDO;
483 			} else if (boton.equals("Espera de comida")) {
484 				estado = Estados.ESPERA_COMIDA;
485 			} else if (boton.equals("Servidos")) {
486 				estado = Estados.SERVIDOS;
487 			} else if (boton.equals("Espera cuenta")) {
488 				estado = Estados.ESPERA_CUENTA;
489 			} else if (boton.equals("Pagando")) {
490 				estado = Estados.PAGANDO;
491 			} else if (boton.equals("Preparación")) {
492 				estado = Estados.PREPARACION;
493 			}
494 			try {
495 				Timestamp timestamp = new Timestamp(
496 						System.currentTimeMillis());
497 				mesa.setHoraEstado(sdf.format(timestamp));
498 
499 				mesa.setEstado(estado);
500 				mesa.modificar();
501 				volverColorInicial();
502 				estadoBotones(estado);
503 
504 			} catch (Exception arg0) {
505 				System.out.println(arg0.getMessage());
506 			}
507 
508 		}
